Commands.Players.Who.Description:		Shows who is online
Commands.Players.Who.Usage:				Usage: '/who <all|nf|name|guild|class|location|level|language>'
Commands.Players.Who.Usage.All:			'/who all'        # Lists all players online.
Commands.Players.Who.Usage.NF:			'/who nf'         # Lists all players online in New Frontiers.
Commands.Players.Who.Usage.CSR:			'/who csr'        # Lists all Customer Service Representatives currently online.
Commands.Players.Who.Usage.DEV:			'/who dev'        # Lists all Development Team Members currently online.
Commands.Players.Who.Usage.QTA:			'/who qta'        # Lists all Quest Team Assistants currently online.
Commands.Players.Who.Usage.Name:		'/who <name>'     # Lists players with names that start with <name>.
Commands.Players.Who.Usage.GuildName:	'/who <guildName> # Lists players with names that start with <guildName>.
Commands.Players.Who.Usage.Class:		'/who <class>'    # Lists players with of class <class>.
Commands.Players.Who.Usage.Location:	'/who <location>' # Lists players in the <location> area.
Commands.Players.Who.Usage.Level:		'/who <level>'    # Lists players of level <level>.
Commands.Players.Who.Usage.LevelLevel:	'/who <level#> <level#> # Lists players in level range.
Commands.Players.Who.Usage.Language:	'/who <language>  # List players with a specific language.
